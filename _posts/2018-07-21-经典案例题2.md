        age = 10
    else:

```

```
def encry(num):
    numstr = ''
    for i in num:
        i = (int(i) + 5) % 10
        numstr += str(i)
    newnum = numstr[::-1]
    print(int(newnum))
if __name__ == "__main__":
    
    num = input('请输入四位的一个整数')
